  8 # Import class.
  9 import sys
 10 import numpy as np
 11 import pandas as pd
 12 import matplotlib as mpl
 13 import matplotlib.pyplot as plt
 14 import statsmodels.api as sm
 15 import statsmodels.robust.norms as norms
 16
 17 # import weights.
 18 from pyamr.datasets.load import make_timeseries
 19 from pyamr.core.regression.wls import WLSWrapper
 20 from pyamr.metrics.weights import SigmoidA
 21
 22 # ----------------------------
 23 # set basic configuration
 24 # ----------------------------
 25 # Matplotlib options
 26 mpl.rc('legend', fontsize=6)
 27 mpl.rc('xtick', labelsize=6)
 28 mpl.rc('ytick', labelsize=6)
 29
 30 # Set pandas configuration.
 31 pd.set_option('display.max_colwidth', 14)
 32 pd.set_option('display.width', 150)
 33 pd.set_option('display.precision', 4)
 34
 35 # ----------------------------
 36 # create data
 37 # ----------------------------
 38 # Create timeseries data
 39 x, y, f = make_timeseries()
 40
 41 # -----------------------------
 42 # Example II
 43 # -----------------------------
 44 # This example performs grid search on a number of possible configurations
 45 # of the WLSWrapper. In particular, it tests the effect of different
 46 # objects to compute the weights from the frequencies. It presents both
 47 # the resulting pandas dataframe and also a figure.
 48
 49 # Configuration
 50 # -------------
 51 # This variable contains the weight functions to test. Note that in
 52 # the norms module there are other options such as [norms.HuberT(),
 53 # norms.Hampel(), norms.TrimmedMean(), norms.TukeyBiweight(),
 54 # norms.AndreWave(), norms.RamsayE()]
 55 w_func = [
 56     norms.LeastSquares(),
 57     SigmoidA(r=200, g=0.5, offset=0.0, scale=1.0),
 58     SigmoidA(r=200, g=0.5, offset=0.0, scale=1.0, percentiles=[10, 90]),
 59     SigmoidA(r=200, g=0.5, offset=0.0, scale=1.0, percentiles=[25, 75]),
 60     SigmoidA(r=200, g=0.5, offset=0.0, scale=1.0, percentiles=[25, 90]),
 61     SigmoidA(r=200, g=0.5, offset=0.0, scale=1.0, percentiles=[40, 50])]
 62
 63 # The grid search parameters.
 64 grid_params = [
 65     # {'exog': [x], 'endog': [y], 'trend': ['c']},
 66     {'exog': [x], 'endog': [y], 'trend': ['c'], 'weights': [f], 'W': w_func}
 67 ]
 68
 69 # Grid search
 70 # ------------
 71 # Perform grid search.
 72 summary = WLSWrapper(estimator=sm.WLS) \
 73     .grid_search(grid_params=grid_params)
 74
 75 # Show grid results
 76 # ..todo: It is weird to create an WLSWrapper jut to
 77 #         be able to use themethod from_list_dataframe.

 79 print("\nGrid search:")

 81

 83 # ----------

101     # Compute predictions.
102     preds = model.get_prediction(start=start, end=end)
103
104     # Plot forecasted values.
105     axes[0].plot(preds[0, :], preds[1, :],
106                  linewidth=1.0,
107                  label=model._identifier(short=True))
108
109     # Plot the confidence intervals.
110     axes[0].fill_between(preds[0, :],
111                          preds[2, :],
112                          preds[3, :],
113                          alpha=0.1)
114
115     # Plot weights assigned to each observation
116     axes[1].plot(model.weights, marker='o', alpha=0.5,
117                  markeredgecolor='k', markeredgewidth=0.5,
118                  markersize=4, linewidth=0.00,
119                  label=model._identifier(short=True))
120
121     # Plot weights converter (W) functions.
122     if model.W is not None:
123         axes[2].plot(np.linspace(0, 1, 100),
124                      model.W.weights(np.linspace(0, 1, 100)),
125                      label=model._identifier(short=True))
126
